# Run ping, DNS, ports, traceroute — one command
param(
  [string]$Target     = "google.com",
  [switch]$SaveReport
)

# ── PING TEST ─────────────────────────────────
$pingResults = Test-Connection $Target -Count 4
$avg = [math]::Round(($pingResults | Measure-Object ResponseTime -Average).Average, 1)
Write-Result "Avg Latency" "$avg ms"

# ── PORT CHECK ────────────────────────────────
foreach ($p in $ports) {
  $tcp = Test-NetConnection -ComputerName $Target -Port $p.Port
  Write-Result "$($p.Name)" $(if ($tcp.TcpTestSucceeded) { "✅ Open" } else { "❌ Closed" })
}
